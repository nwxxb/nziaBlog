```bash
# please use this command to make the drafted/unpublished/example content visible
hugo server -D

# generators
# generate a single-file project with no notes/scracthpad
# this will generate content/projects/<project name>.md
hugo new projects/<project name>.md

# generate a project that has notes/scracthpad (a leaf-bundle project)
# this will generate content/projects/<project name>/_index.md and some markdown notes
hugo new projects/<project name>

# generate a post
# this will generate content/posts/<post name>.md
hugo new posts/<project name>.md
```
